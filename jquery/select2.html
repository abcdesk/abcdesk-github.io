    }
});

function getTopicList(element) {
    var topicList = ''
    $.ajax({
        type:'get',
        contentType:'application/json;charset=utf-8',
        url: '/api/topics',
        dataType:'json',
        success: function(data) {
            if(status == 00) {
                $.each(data, function(i, item){
                    topicList += ''
                });
                element.html(topicList)
            }
        }
    })
}
